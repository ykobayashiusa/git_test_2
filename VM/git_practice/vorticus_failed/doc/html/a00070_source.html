24 panel_strategy<eorder,gorder,aorder>::panel_strategy( const grid_t& g ):
25 elem_matrices ( g.size() ), partial_results ( g.size() )
26 {
27  std::vector<const triangle<gorder>*> trias;
28 
29  // Create all entries.
30  for ( const triangle<gorder>& t: g )
31  {
32  elem_matrices [ t.pos() ];
33  partial_results[ t.id() ];
34  trias.push_back( &t );
35  }
36 
37  // Compute the self-interactions.
38  #pragma omp parallel for
39  for ( size_t i = 0; i < trias.size(); ++i )
40  {
41  const triangle<gorder>& t = *trias[ i ];

43  }

46 template <uint eorder, uint gorder, uint aorder>
47 void panel_strategy<eorder,gorder,aorder>::apply( const grid_t& grid,
48  const grid_function<real,gorder,aorder>& source,
49  grid_function<real,gorder,aorder>& target )
50 {
51  source_strengths = &source;
52 
53  for ( auto& value_pair: target )
54  {
55  value_pair.second = 0;
56  }
57 
58  for ( auto& tmp: partial_results )
59  {
60  auto& vals = tmp.second;
61  for ( size_t k = 0; k < vals.size(); ++k )
62  {
63  vals[ k ] = 0;
64  }
65  }
66 
67  fmm::fmm( *this, grid.begin(), grid.end(), grid.begin(), grid.end() );
68 
69  for ( const auto& tmp: partial_results )
70  {
71  const auto& vals = tmp.second;
72  const triangle<gorder>& t = grid.get_multi_grid().get_triangle( tmp.first );
73  const auto& dof_map = target.get_dofs( t );
74 
75  for ( size_t k = 0; k < vals.size(); ++k )
76  {
77  target( dof_map[ k ] ) += vals[ k ];
78  }
79  }

82 template <uint eorder, uint gorder, uint aorder>
83 void panel_strategy<eorder,gorder,aorder>::p2m
84 (
85  Mcoeff_t &M, point xc, const source_iter *const begin,
86  const source_iter *const end
87 ) const noexcept
88 {
89  using math::F;
90  using shapefcts2d::N;
91  using shapefcts2d::apply;
92  using geometry::cmplx_point;
93 
94 
95  const uint quad_order = eorder + aorder + 1;
96  const uint quad_num = dunavant_num_points[ quad_order ];
97  const bary *quad_points = dunavant_points [ quad_order ];
98  const real *quad_weights = dunavant_weights[ quad_order ];
99 
100  for ( const source_iter *i = begin; i != end; ++i )
101  {
102  const triangle<gorder>& t = **i;
103  const auto dof_map = source_strengths->get_dofs( t );
104 
105  std::array<real,dof_map.size()> dofs;
106  for ( size_t j = 0; j < dofs.size(); ++j )
107  {
108  dofs[ j ] = (*source_strengths)( dof_map[j] );
109  }
110 
111  for ( uint q = 0; q < quad_num; ++q )
112  {
113  const bary pos = quad_points [ q ];
114  const real w = quad_weights[ q ];
115 
116  const point yy = t.Chi( pos );
117  const real gy = t.surf_elem( pos );
118 
119  const auto phi = rotN<gorder,aorder>( t, pos );
120  const auto f = F<eorder>( yy - xc );
121 
122  const auto val = apply<aorder,point>( phi, dofs );
123 
124  for ( size_t k = 0; k < M.num_coeff(); ++k )
125  M.data[k] += 0.5*w*gy*f.data[k]*val;
126  }
127  }
128 }

131 void panel_strategy<eorder,gorder,aorder>::m2m( Mcoeff_t &target, const Mcoeff_t &source, point r ) const noexcept
132 {
133  using math::F;
134  using math::idx;
135 
136  auto transfer = F<eorder>( r );
137 
138  for ( int n = 0; n < eorder; ++n )
139  {
140  for ( int m = 0; m <= n; ++m )
141  {
142  for ( int nd = 0; nd <= n; ++nd )
143  {
144  int min_md = std::max(-nd, m - (n-nd));
145  int max_md = std::min( nd, m + (n-nd));
146  for ( int md = min_md; md <= max_md; ++md )
147  {
148  target.data[ idx(n,m) ] += source( nd, md )*conj(transfer(n-nd,m-md));
149  }
150  }

308 template <uint eorder, uint gorder, uint aorder>
309 void panel_strategy<eorder,gorder,aorder>::p2p( target_iter *const tbegin, target_iter *const tend,
310  const source_iter *const sbegin, const source_iter *const send ) noexcept
311 {
312  arma::vec::fixed< shapefcts2d::num<aorder>() > rhs;
313 
314  for ( target_iter *i = tbegin; i != tend; ++i )
315  {
316  const triangle<gorder> &t1 = **i;
317  const auto dofs1 = source_strengths->get_dofs( t1 );
318 
319  for ( const source_iter *j = sbegin; j != send; ++j )
320  {
321  const triangle<gorder> &t2 = **j;
322  const auto dofs2 = source_strengths->get_dofs( t2 );
323 
324  const elem_mat_t& elem_mat = get_elem_mat( t1, t2 );
325 
326  for ( size_t k = 0; k < dofs2.size(); ++k )
327  {
328  rhs( k ) = (*source_strengths)( dofs2[ k ] );
329  }
330 
331  rhs = elem_mat*rhs;
332 
333  shapefcts2d::vals<aorder>& target = partial_results[ t1.id() ];
334  for ( size_t k = 0; k < dofs1.size(); ++k )
335  {
336  target[ k ] += rhs(k);
337  }
338  }
339  }

349 template <uint eorder, uint gorder, uint aorder>
350 auto panel_strategy<eorder,gorder,aorder>::get_elem_mat( const triangle<gorder> &t1,
351  const triangle<gorder> &t2 )
352 -> const elem_mat_t&
353 {
354  if ( elem_matrices[ t1.pos() ].find( t2.pos() ) == elem_matrices[ t1.pos() ].end() )
