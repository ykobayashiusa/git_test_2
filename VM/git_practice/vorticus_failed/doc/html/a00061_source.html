23 template <typename T, uint gorder, uint aorder>
24 grid_function<T,gorder,aorder>::grid_function( const grid<gorder> &g ):
25 level_ { g.get_level() }
26 {
27  using std::make_pair;
28 
29  coords_.reserve( g.size() );
30  const T default_value {};
31  for ( const auto &t: g )
32  {
33  coords_t& map = coords_[ t.id() ];
34 
35  // Corner dofs.
36  values_.insert( make_pair(t.get_node(0), default_value) ); map[ 0 ] = t.get_node(0);
37  values_.insert( make_pair(t.get_node(1), default_value) ); map[ 1 ] = t.get_node(1);
38  values_.insert( make_pair(t.get_node(2), default_value) ); map[ 2 ] = t.get_node(2);
39 
40  // Edge dofs.
41  for ( unsigned char e_num { 0 }; e_num < 3; ++e_num )
42  {
43  const edge<gorder>& e { t.get_edge(e_num) };
44 
45  size_t offset;
46  switch ( e_num )
47  {
48  case 0: offset = aorder + 1; break;
49  case 1: offset = 2*aorder + 0; break;
50  case 2: offset = 2; break;
51  }
52 
53  bool reverse;
54  if ( (e.has_triangle(0) && &e.get_triangle(0) == &t) ||
55  (e.has_triangle(2) && &e.get_triangle(2) == &t) )
56  {
57  reverse = false;
58  }
59  else
60  {
61  reverse = true;
62  }
63 
64  for ( size_t k { 1 }; k < aorder; ++k )
65  {
66  const real pos { ((real) k)/((real) aorder) };
67  values_.insert( make_pair(e.Chi(pos), default_value) );
68 
69  const size_t tria_dof_no = reverse ? offset+(aorder-k) : offset+k;
70  map[ tria_dof_no ] = e.Chi(pos);
71  }
72  }
73 
74  // Inner dofs.
75  constexpr auto pos = lattice::lattice<aorder>();
76  for ( size_t k { 3*aorder }; k < lattice::size<aorder>(); ++k )
77  {
78  values_.insert( make_pair(t.Chi(pos[k]), default_value) );
79  map[ k ] = t.Chi(pos[k]);
80  }
81  }
82 }

85 T grid_function<T,gorder,aorder>::eval( const triangle<gorder>& t, bary pos ) const
86 {
87  using namespace shapefcts2d;
88  coeffs<aorder,T> coeffs;
89  const coords_t& map = coords_.find( t.id() )->second;
90 
91  for ( size_t i { 0 }; i < map.size(); ++i )
92  {
93  coeffs[ i ] = values_.find( map[ i ] )->second;

202 template <uint gorder, uint aorder>
203 void prolongation<gorder,aorder>::treat_triangle( const triangle<gorder>& t,
204  const scalar& coarse, const scalar& fine )
205 {
206  const auto ref_status = t.get_ref_status();
207 
208  switch ( ref_status.count() )
209  {
210  case 0: treat_no_ref ( t, coarse, fine ); break;
211  case 1: treat_single_irreg_ref( t, coarse, fine ); break;
212  case 2: treat_double_irreg_ref( t, coarse, fine ); break;
213  case 3: treat_regular_ref ( t, coarse, fine ); break;
214  }

232 void prolongation<gorder,aorder>::treat_single_irreg_ref( const triangle<gorder>& t,
233  const scalar &coarse, const scalar &fine )
234 {
235  const auto& coarse_map = coarse.coords_.find( t.id() )->second;
236  const auto ref_status = t.get_ref_status();
237  const auto& fine_map_0 = fine.coords_.find( t.get_child_id( 0 ) )->second;
238  const auto& fine_map_1 = fine.coords_.find( t.get_child_id( 1 ) )->second;
239 
240  bary t0_0, t0_1, t0_2;
241  bary t1_0, t1_1, t1_2;
242  if ( ref_status.test(0) )
243  {
244  t0_0 = bary { 1, .0, .0 }; t0_1 = bary { .0, 1, .0 }; t0_2 = bary { .0, .5, .5 };
245  t1_0 = bary { .0, .0, 1 }; t1_1 = bary { 1, .0, .0 }; t1_2 = bary { .0, .5, .5 };
246  }
247  else if ( ref_status.test(1) )
248  {
249  t0_0 = bary { 1, .0, .0 }; t0_1 = bary { .0, 1, .0 }; t0_2 = bary { .5, .0, .5 };
250  t1_0 = bary { .0, 1, .0 }; t1_1 = bary { .0, .0, 1 }; t1_2 = bary { .5, .0, .5 };
251  }
252  else // ( ref_status.test(2) )
253  {
254  t0_0 = bary { .0, .0, 1 }; t0_1 = bary { 1, .0, .0 }; t0_2 = bary { .5, .5, .0 };
255  t1_0 = bary { .0, 1, .0 }; t1_1 = bary { .0, .0, 1 }; t1_2 = bary { .5, .5, .0 };
256  }
257 
258  constexpr auto positions = lattice::lattice<aorder>();
259  for ( size_t k { 0 }; k < positions.size(); ++k )
260  {
261  const bary pos = positions[ k ];
262  const bary pos0 { pos.z0*t0_0 + pos.z1*t0_1 + pos.z2*t0_2 };
263  const bary pos1 { pos.z0*t1_0 + pos.z1*t1_1 + pos.z2*t1_2 };
264 
265  using shapefcts2d::N;
266  const auto vals0 = N<aorder>( pos0 );
267  const auto vals1 = N<aorder>( pos1 );
268 
269  for ( size_t j { 0 }; j < positions.size(); ++j )
270  {
271  const idx_t idx0 { coarse_map[ j ], fine_map_0[ k ] };
272  const idx_t idx1 { coarse_map[ j ], fine_map_1[ k ] };
273 
274  matrix_[ idx0 ] = vals0[ j ];
275  matrix_[ idx1 ] = vals1[ j ];
276  }
277  }
