91  point Bcentre, real Bsize ) const
92 {
93  const real dist = length(Acentre-Bcentre) - (Asize+Bsize)/2;

102  const source_iter *const begin,
103  const source_iter *const end ) const
104 {
105  using math::dF;
106  for ( auto pp = begin; pp != end; ++pp )

120 {

122  using math::idx;
123 
124  auto transfer = F<order>( r );
125 
126  for ( int n = 0; n < order; ++n )
127  {
128  for ( int m = 0; m <= n; ++m )
129  {
130  for ( int nd = 0; nd <= n; ++nd )
131  {
132  int min_md = std::max(-nd, m - (n-nd));
133  int max_md = std::min( nd, m + (n-nd));
134  for ( int md = min_md; md <= max_md; ++md )
135  {
136  target.data[ idx(n,m) ] += source( nd, md )*conj(transfer(n-nd,m-md));

217 template <uint order>
218 void particle_strategy<order>::p2p( target_iter *const tbegin, target_iter *const tend,
219  const source_iter *const sbegin, const source_iter *const send ) const
220 {
221  using geometry::tensor;
222  using std::exp;
223 
224  for ( auto tt = tbegin; tt != tend; ++tt )
225  {
226  target_iter t = *tt;
227  for ( auto ss = sbegin; ss != send; ++ss )
228  {
229  source_iter s = *ss;
230  const point R = t->x - s->x;
231 
232  real r = R.r();
233  real sm1 = 1/sigma; sm1 = std::isnormal(sm1) ? sm1 : 0;
234  real rm1 = 1 / r; /*rm1 = std::isnormal(rm1) ? rm1 : 0;*/ rm1 = r < 1e-6 ? 0 : rm1;
235 
236  real rs = r*sm1;
237  real rs3 = rs*rs*rs;
238  real sm3 = sm1*sm1*sm1;
239  real rm2 = rm1*rm1;
240  real rm3 = rm2*rm1;
241  real rm5 = rm3*rm2;
242 
243  real G = -expm1(-rs3);
244  real GD = rm2*sm3*exp(-rs3);
245 
246  tensor tmp = dyad_prod( R, R );
247  tmp *= 3*(rm5*G - GD);
248  tmp(0,0) -= rm3*G; tmp(1,1) -= rm3*G; tmp(2,2) -= rm3*G;
249 
250  t->u += cross_prod( s->G, R*(-rm3*G) )*pifac;
251  t->du += cross_prod( s->G, tmp )*pifac;
252  }
253  }

290 
291  misc::random_real rand(-1,1);
292  for ( uint i = 0; i < n; ++i )
293  {
294  data[ i ].x = geometry::point( rand(), rand(), rand() );
295  data[ i ].G = geometry::point( rand(), rand(), rand() );
296  }
297 
298  geometry::point r( rand(), rand(), rand() );
299 
300  misc::stopwatch clock;
301 
302  clock.reset();
303  p2p( begin, end, begin, end );
304  time_p2p_ = clock.elapsed() / (n*n);
305 
306  Mcoeff_t M;
307  Lcoeff_t L;
308 
309  clock.reset();
310  m2p( begin, end, L, r );
311  time_m2p_ = clock.elapsed() / n;

315  time_m2l_ = clock.elapsed();
