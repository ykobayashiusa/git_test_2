239 void neighbourhood_finder::find_neighbourhood( const particle &p, real radius,
240  std::vector<particle*>& target )
241 {
242  using geometry::point;
243  using geometry::length;
244 
245  point centre = p.x;
246  point min;
247  min.x = centre.x - radius;
248  min.y = centre.y - radius;
249  min.z = centre.z - radius;
250 
251  point max;
252  max.x = centre.x + radius;
253  max.y = centre.y + radius;
254  max.z = centre.z + radius;
255 
256  std::vector<particle*> results;
257  this->query( min, max, results );
258 
259  for ( int i = 0; i < results.size(); ++i )
260  {
261  const point &pp = results[ i ]->x;
262  if ( length( pp - centre ) < radius )
263  {
264  particle &candidate = *results[ i ];
265  if ( &candidate != &p )
266  target.push_back( &candidate );
267  }
268  }
269 }
