31 wigner_matrix::wigner_matrix( int l, real theta ):
32  _l( l ), _data( 2*l+1, 2*l+1 )
33 {
34  for ( int m = -l; m <= l; ++m )
35  {
36  _data( idx(m,l) ) = g_lm( l, m )*
37  pow(1+cos(theta),m)*
38  pow(sin(theta),l-m);
39  }
40 
41  for ( int k = l - 1; k >= -l; --k )
42  {
43  _data( idx(l,k) ) = -((l + k + 1.0) /
44  sqrt(l*(l+1.0)-k*(k+1.0)))*
45  ((sin(theta))/(1+cos(theta)))*
46  _data( idx(l,k+1) );
47  }
48

50  {

52  {
53  _data( idx(m,k) ) = sqrt( (l*(l+1.0)-m*(m+1.0))/
54  (l*(l+1.0)-k*(k+1.0)) )*
55  _data( idx(m+1,k+1) );
56  _data( idx(m,k) )-= ((m + k + 1.0) /
57  sqrt(l*(l+1.0)-k*(k+1.0)))*
58  ((sin(theta))/(1+cos(theta)))*
59  _data( idx(m,k+1) );
60  }
61  }

85 real g_lm( int l, int m )
86 {
87  real result = 1;
88 
89  while ( m > 0 )
90  {
91  result *= std::sqrt( (l-m+1.0)/(l+m) );
92  --m;
93  }
94  while ( l > 0 )
95  {
96  result *= std::sqrt( (2.0*l - 1.0)/(2.0*l) );
97  --l;
98  }
99  return result;
100 }
