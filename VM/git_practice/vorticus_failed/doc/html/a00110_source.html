188 template <uint order>
189 real_coeff<order> legendre_pnm( const real x )
190 {
191  using std::sqrt;
192 
193  real_coeff<order> result;
194  real *const data = result.data.begin();
195 
196  // Starting point of the recurrence.
197  data[ idx( 0, 0 ) ] = 1;
198 
199  // First do the diagonal.
200  for ( uint m = 0; m < order - 1; ++m )
201  {
202  data[ idx(m+1,m+1) ] = data[ idx(m,m) ] *
203  (-1)*(2*m+1)*sqrt( 1 - x*x );
204 
205  }
206 
207  // Afterwards the "small diagonal" below.
208  for ( uint m = 0; m < order - 1; ++m )
209  {
210  data[ idx(m+1,m) ] = data[ idx(m,m) ]*x*(2*m+1);
211  }
212 
213  // Now the rest.
214  for ( uint m = 0; m < order - 2; ++m )
215  {
216  for ( uint n = m + 2; n < order; ++n )
217  {
218  data[ idx(n,m) ] = ( data[ idx(n-1,m) ]*x*(2*n-1) -
219  data[ idx(n-2,m) ]*(n+m-1) ) /
220  ( n - m );
221  }
222  }
223 
224  return result;
225 }

227 template <uint order>
228 real_coeff<order> differentiate( const real_coeff<order> &pnm )
229 {
230  real_coeff<order> result {pnm};
231  real *const data {result.data.begin()};
232 
233  const real x {data[ idx(1,0) ]};
234  const real frac = 1.0/(x*x-1);
235 
236  for ( uint n = order - 1; n > 0; --n )
237  {
238  for ( uint m = 0; m < n; ++m )
239  {
240  data[idx(n,m)] = frac*( x*n*data[idx(n, m)] -
241  (n+m)*data[idx(n-1,m)]);
242  }
243  data[idx(n,n)] = frac*x*n*data[idx(n,n)];
244  }
245  data[idx(0,0)] = 0;
246 
247  return result;
248 }

250 template <uint order>
251 cmplx_coeff<order> F( const geometry::point p_r )
252 {
253  using std::sqrt;
254  using std::exp;
255 
256  const real r { p_r.r() };
257  const real costheta { p_r.z / p_r.r() };
258  const real phi { p_r.phi() };
259 
260  const auto pnm = legendre_pnm<order>( costheta );
261 
262  real rn {1}; // r^n
263  real nfac {1}; // n!
264 
265  cmplx_coeff<order> result;
266  for ( uint n = 0; n < order; ++n )
267  {
268  real nmfac {nfac}; // (n+m)!
269 
270  for ( uint m = 0; m <= n; ++m )
271  {
272  real val { rn*( pnm.data[ idx(n,m) ] )/nmfac };
273  if ( (n-m) % 2 ) val = -val;
274  result.data[ idx(n,m) ] = val*exp( cmplx(0,m*phi) );
275 
276  nmfac *= n + m + 1;
277  }
278 
279  rn *= r;
280  nfac *= n + 1;
281  }
282 
283  return result;
284 }

286 template<uint order>
287 cmplx_point_coeff<order> dF( const geometry::point p_r )
288 {
289  using geometry::point;
290  using geometry::cmplx_point;
291  using std::sqrt;
292  using std::sin;
293  using std::cos;
294  using std::exp;
295 
296  const real r { p_r.r() };
297  const real phi { p_r.phi() };
298  const real theta { p_r.theta() };
299  const real sintheta { sin(theta) };
300 
301  const point e_r = p_r.e_r();
302  const point e_theta = p_r.e_theta();
303  const point e_phi = p_r.e_phi();
304 
305  const auto pnm = legendre_pnm<order>( cos(theta) );
306  const auto d_pnm = differentiate(pnm);
307 
308  real rn {1}; // r^n
309  real nfac {1}; // n!
310 
311  cmplx_point_coeff<order> result;
312  for ( uint n = 0; n < order; ++n )
313  {
314  real nmfac = nfac; // (n+m)!
315  for ( uint m = 0; m <= n; ++m )
316  {
317  const cmplx dF_dr { ((rn/r)/nmfac)*n*exp(cmplx(0,m*phi))*pnm.data[ idx(n,m) ] };
318  const cmplx dF_dtheta { (rn/nmfac)*exp(cmplx(0,m*phi))*d_pnm.data[ idx(n,m) ]*(-sintheta) };
319  const cmplx dF_dphi { (rn/nmfac)*cmplx(0,m)*exp(cmplx(0,m*phi))*pnm.data[ idx(n,m) ] };
320 
321  result.data[ idx(n,m) ] = dF_dr*e_r +
322  (dF_dtheta/r)*e_theta +
323  (dF_dphi/(r*sintheta))*e_phi;
324 
325  if ( (n-m) % 2 ) result.data[ idx(n,m) ] *= -1;
326  nmfac *= n + m + 1;
327  }
328 
329  rn *= r;
330  nfac *= n + 1;
331  }
332 
333  return result;
334 }
