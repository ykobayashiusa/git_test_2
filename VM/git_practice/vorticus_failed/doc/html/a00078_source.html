22 template <uint eorder, uint gorder, uint aorder>
23 solver<eorder,gorder,aorder>::solver( multigrid<gorder>& mg ):
24 mg_ (mg)
25 {
26  for ( size_t l = 0; l <= mg.last_level(); ++l )
27  {
28  grids_.push_back( grid(mg,l) );
29  strategies_.push_back( strategy(grids_[l]) );
30  defects_.push_back( fct( grids_[l] ) );
31 
32  if ( l >= 1 )
33  prols_.push_back( prol( grids_[ l - 1 ], grids_[ l ] ) );
34 
35  numberings_.push_back( std::unordered_map<point,size_t>() );
36  auto& numb = numberings_.back();
37  size_t counter { 0 };
38  for ( const auto& p: defects_[ l ] ) numb[ p.first ] = counter++;
39 
40  diagonals_.push_back( math::vector( numb.size(), arma::fill::zeros ) );
41  auto& diag = diagonals_[ l ];
42 
43  for ( const auto& t: grids_[l] )
44  {
45  const auto& elem_mat = strategies_[l].get_elem_mat( t, t );
46  const auto& dofs = defects_[l].get_dofs( t );
47 
48  for ( size_t i = 0; i < elem_mat.n_rows; ++i )
49  diag( numb[ dofs[ i ] ] ) += elem_mat( i, i );
50  }
51  }
52 }

54 template <uint eorder, uint gorder, uint aorder>
55 auto solver<eorder,gorder,aorder>::compute_rhs( const fct& g_N ) -> fct
56 {
57  unsigned char level = g_N.get_level();
58  math::vector result( numberings_[level].size(), arma::fill::zeros );
59  for ( const auto& t: grids_[level] )
60  {
61  const math::vector elem_vec = rhs<gorder,aorder>(t);
62  const auto& dofs = g_N.get_dofs( t );
63 
64  for ( size_t i = 0; i < elem_vec.n_rows; ++i )
65  result( numberings_[ level ][ dofs[ i ] ] ) -= elem_vec( i );
66  }
67 
68  return convert(level,result);
69 }

103 template <uint eorder, uint gorder, uint aorder>
104 void solver<eorder,gorder,aorder>::solve( fct& sol, const fct& g_N )
105 {
106  const fct rhs = compute_rhs( g_N );
107  unsigned char level = rhs.get_level();
108 
109  // Initialise defect.
110  if ( level != 0 )
111  {
112  strategies_[ level ].apply( grids_[ level ], sol, defects_[ level ] );
113  for ( auto& val: defects_[ level ] ) val.second -= rhs( val.first );
114  }
115 
116  real residual { 0 }; size_t counter { 0 };
117  do
118  {
119  residual = mgm( sol, rhs );
120  std::cout << "Multigrid-iteration " << ++counter
121  << ". Residual: " << residual << "." << std::endl;
122  }
123  while ( residual > std::pow( 2., -3*level ) && counter < 20 );
124 }

126 template <uint eorder, uint gorder, uint aorder>
127 real solver<eorder,gorder,aorder>::mgm( fct& sol, const fct& rhs )
128 {
129  constexpr size_t nu1 = 3; // Number of pre-smoothing iterations.
130  constexpr size_t nu2 = 0; // Number of post-smoothing iterations.
131 
132  unsigned char level = rhs.get_level();
133 
134  if ( level == 0 )
135  {
136  cg( sol, rhs );
137  return 0;
138  }
139 
140  auto& S = strategies_[ level ];
141 
142  // These values will get overwritten, but copy construction
143  // should be way more faster than construction from scratch.
144  fct d_coarse = defects_[ level - 1 ];
145  fct c_coarse = defects_[ level - 1 ];
146 
147  for ( size_t i { 0 }; i < nu1; ++i )
148  smooth( sol, rhs );
149 
150  prols_[ level - 1 ].restrict( d_coarse, defects_[ level ] );
151 
152  // We don’t have a meaningful starting value for c_coarse,
153  // starting from zero allows us to easily initialise the
154  // lower level defect vector.
155  for ( auto& val: c_coarse ) val.second = 0;
156  for ( auto& val: defects_[ level - 1 ] ) val.second = - d_coarse( val.first );
157 
158  mgm( c_coarse, d_coarse );
159 
160  fct c_fine = defects_[ level ];
161  prols_[ level - 1 ].prolongate( c_coarse, c_fine );
162 
163  for ( auto& val: sol )
164  val.second -= c_fine( val.first );
165 
166 
167  // Reinitialise defect.
168  S.apply( grids_[ level ], sol, defects_[ level ] );
169  real residual { 0 };
170  for ( auto& val: defects_[ level ] )
171  {
172  val.second -= rhs( val.first );
173  residual += val.second*val.second;
174  }
175  residual = std::sqrt(residual);
176 
177  for ( size_t i { 0 }; i < nu2 && residual < 1e-9; ++i )
178  residual = smooth( sol, rhs );
179 
180  return residual;
181 }

183 template <uint eorder, uint gorder, uint aorder>
184 real solver<eorder,gorder,aorder>::smooth( fct& sol, const fct& rhs )
185 {
186  unsigned char level = rhs.get_level();
187 
188  auto& S = strategies_[ level ];
189  auto& numb = numberings_[ level ];
190  auto& diag = diagonals_ [ level ];
191  auto& d = defects_ [ level ];
192 
193  for ( auto& p: sol )
194  p.second -= (2./3.) * ( d(p.first) / diag(numb[p.first]) );
195 
196  S.apply( grids_[ level ], sol, d );
197  real residual { 0 };
198  for ( auto& val: d )
199  {
200  val.second -= rhs( val.first );
201  residual += val.second*val.second;
202  }
203  return std::sqrt( residual );
204 }

206 template <uint eorder, uint gorder, uint aorder>
207 void solver<eorder,gorder,aorder>::cg( fct& sol, const fct& rhs )
208 {
209  const unsigned char lvl = rhs.get_level();
210  auto& S = strategies_[ lvl ];
211  auto& diag = diagonals_[ lvl ];
212 
213  const math::vector b = convert( rhs );
214  math::vector x = convert( sol );
215 
216 
217  fct zz(sol);
218  fct dd(sol);
219 
220  real alpha, beta;
221  math::vector r, h, d, z;
222 
223  r.zeros( b.size() );
224  h.zeros( b.size() );
225  d.zeros( b.size() );
226  z.zeros( b.size() );
227 
228  S.apply( grids_[ lvl ], dd, zz );
229  for ( auto& val: zz )
230  {
231  val.second = rhs( val.first ) - val.second;
232  }
233 
234  r = convert( zz );
235  h = r / diag;
236 
237  d = h;
238  dd = convert( lvl, d );
239 
240  for ( size_t k = 0; k <= 100; ++k )
241  {
242  S.apply( grids_[ lvl ], dd, zz );
243  z = convert( zz );
244 
245  alpha = dot( r, h ) / dot( d, z );
246 
247  math::vector rold = r;
248  math::vector hold = h;
249 
250  x = x + alpha*d;
251  r = r - alpha*z;
252  h = r / diag;
253 
254  if ( norm(r) < 1e-9 || k == 100 )
255  break;
256 
257  beta = dot( r, h )/dot( rold, hold );
258  d = h + beta*d;
259  dd = convert( lvl, d );
260  }
261  sol = convert( lvl, x );
262 }
