76 int simplex::solve( real *const solution )
77 {
78  typedef arma::Mat<real> mat;
79  using arma::eye;
80 
81  typedef arma::Col<real> vec;
82  using arma::uword;
83  using arma::uvec;
84  using arma::linspace;
85 
86  // Ensure that the right hand side is positive.
87  preprocess();
88 
89  mat A_mat( A, constraints, constraints + variables, false, true );
90  vec b_vec( b, constraints, false, true );
91 
92  // Create identity matrix for the artificial variables.
93  for ( int i = 0; i < constraints; ++i )
94  A_mat( i, variables + i ) = 1;
95 
96  // Create the artifical objective function.
97  vec c_vec( variables + constraints );
98  c_vec.fill( 0 );
99  c_vec.subvec( variables, variables + constraints - 1 ).fill( -1 );
100 
101  // Build the initial basis.
102  uvec basic_variables = linspace<uvec>( variables, variables + constraints - 1, constraints );
103 
104  // Create the bottom row of the tableau.
105  vec bottom_row( variables + constraints );
106  real corner_val;
107 
108  vec theta( constraints );
109 
110  int iter = 0;
111  while ( true )
112  {
113  bottom_row = c_vec;
114  uvec artificial_rows = find( basic_variables >= variables );
115  for ( int i = 0; i < variables + constraints; ++i )
116  {
117  for ( int j = 0; j < artificial_rows.size(); ++j )
118  {
119  bottom_row( i ) += A_mat( artificial_rows(j), i );
120  }
121  }
122  bottom_row( basic_variables ).fill(0);
123 
124  corner_val = 0;
125  for ( int j = 0; j < artificial_rows.size(); ++j )
126  {
127  corner_val -= b_vec( artificial_rows( j ) );
128  }
129 
131  // Pivot selection.
133  uword pivot_col; real val = bottom_row.max( pivot_col );
134  if ( val <= 0 )
135  {
136  // Optimal solution has been found.
137  break;
138  }
139 
140  uword pivot_row;
141 
142  if ( iter > 200 )
143  {
144  // Use Bland's rule for a couple of iterations if we
145  // already have iterated too much.
146  uvec candidates = find( bottom_row > 0 );
147  pivot_col = candidates.min();
148 
149  theta = b_vec / A_mat.col( pivot_col );
150  theta( find( A_mat.col(pivot_col) <= 0 ) ).fill( DBL_MAX );
151  val = theta.min();
152 
153  candidates = find( theta == val );
154  uvec numbers = basic_variables( candidates );
155 
156  numbers.min( pivot_row );
157  pivot_row = candidates( pivot_row );
158 
159  if ( iter > 300 ) iter = 0;
160  }
161  else
162  {
163  // The correct pivot column has already been computed
164  // when checking for convergence.
165  theta = b_vec / A_mat.col( pivot_col );
166  theta( find( A_mat.col(pivot_col) <= 0 ) ).fill( DBL_MAX );
167  theta.min( pivot_row );
168  }
169 
170 
172  // Change of basis variables.
174  basic_variables( pivot_row ) = pivot_col;
175 
176  const real pivot = A_mat( pivot_row, pivot_col );
177  A_mat.row( pivot_row ) /= pivot;
178  b_vec( pivot_row ) /= pivot;
179 
180  for ( int i = 0; i < constraints; ++i )
181  {
182  if ( i == pivot_row ) continue;
183 
184  const real tmp = A_mat( i, pivot_col );
185  A_mat.row( i ) -= tmp*A_mat.row( pivot_row );
186  b_vec( i ) -= tmp*b_vec( pivot_row );
187  }
188  ++iter;
189  }
190 
191  if ( std::abs( corner_val ) <= 1e-9 )
192  {
193  // Assume that there is a feasible solution.
194  std::fill( solution, solution + variables, 0 );
195  for ( int i = 0; i < constraints; ++i )
196  {
197  int index = basic_variables[ i ];
198  if ( index < variables )
199  {
200  solution[ index ] = b_vec( i );
201  }
202  }
203 
204  // Indicate a successful solution.
205  return 0;
206  }
207  else
208  {
209  // There is no feasible solution.
210  // Indicate an unsuccessful solution.
211  return 1;
212  }
213 }

215 void simplex::preprocess()
216 {
217  typedef arma::Mat<real> mat;
218  typedef arma::Col<real> vec;
219 
220  mat A_mat( A, constraints, constraints + variables, false, true );
221  vec b_vec( b, constraints, false, true );
222 
223  for ( int i = 0; i < constraints; ++i )
224  {
225  if ( b_vec(i) < 0 )
226  {
227  b_vec(i) *= -1;
228  A_mat.row(i) *= -1;
229  }
230 
231  // Preconditioning.
232  const real val = norm( A_mat.row(i), 1 );
233  A_mat.row(i) /= val;
234  b_vec(i) /= val;
235  }
236 }
